CSS

/* Depth Layers Effect - Multi-layer parallax depth with translateZ */

.depth-layers {
  position: relative;
  display: inline-block;
  overflow: hidden;
  border-radius: 12px;
  perspective: 1000px;
  transform-style: preserve-3d;
}

.depth-layers img {
  display: block;
  width: 100%;
  height: auto;
  transform-style: preserve-3d;
  transition: transform 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94),
              filter 0.3s ease-out;
  will-change: transform;
}

/* Layer 1: Foreground elements (closest to viewer) */
.depth-layers::before {
  content: '';
  position: absolute;
  inset: -10%;
  background: inherit;
  background-size: cover;
  background-position: center;
  transform-style: preserve-3d;
  pointer-events: none;
  border-radius: 12px;
  opacity: 0;
  transition: opacity 0.3s ease-out, transform 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94);
  filter: brightness(1.1) contrast(1.1) saturate(1.2);
  z-index: 10;
  transform: translateZ(80px) scale(0.92);
}

/* Layer 2: Mid-ground */
.depth-layers::after {
  content: '';
  position: absolute;
  inset: -5%;
  background: inherit;
  background-size: cover;
  background-position: center;
  transform-style: preserve-3d;
  pointer-events: none;
  border-radius: 12px;
  opacity: 0;
  transition: opacity 0.3s ease-out, transform 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94);
  filter: brightness(1.05) contrast(1.05);
  z-index: 5;
  transform: translateZ(40px) scale(0.96);
}

/* Layer 3: Background (furthest back) */
.depth-layers .layer-bg {
  position: absolute;
  inset: 0;
  background: inherit;
  background-size: cover;
  background-position: center;
  transform-style: preserve-3d;
  pointer-events: none;
  border-radius: 12px;
  opacity: 0;
  transition: opacity 0.3s ease-out, transform 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94);
  filter: brightness(0.85) blur(4px);
  z-index: -1;
  transform: translateZ(-40px) scale(1.08);
}

/* Layer 4: Deep background */
.depth-layers .layer-deep {
  position: absolute;
  inset: 0;
  background: inherit;
  background-size: cover;
  background-position: center;
  transform-style: preserve-3d;
  pointer-events: none;
  border-radius: 12px;
  opacity: 0;
  transition: opacity 0.3s ease-out, transform 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94);
  filter: brightness(0.7) blur(8px);
  z-index: -2;
  transform: translateZ(-80px) scale(1.15);
}

/* Atmospheric fog/haze between layers */
.depth-layers .atmosphere {
  position: absolute;
  inset: 0;
  background: linear-gradient(
    180deg,
    rgba(255,255,255,0.05) 0%,
    transparent 40%,
    transparent 60%,
    rgba(0,0,0,0.05) 100%
  );
  opacity: 0;
  pointer-events: none;
  transition: opacity 0.4s ease-out;
  z-index: 2;
  border-radius: 12px;
}

/* Parallax indicator */
.depth-layers .parallax-indicator {
  position: absolute;
  inset: 0;
  pointer-events: none;
  z-index: 20;
  opacity: 0;
  transition: opacity 0.3s ease;
}

.depth-layers .parallax-indicator::before {
  content: '';
  position: absolute;
  top: 50%; left: 50%;
  width: 40px; height: 40px;
  border: 2px solid rgba(255,255,255,0.5);
  border-radius: 50%;
  transform: translate(-50%, -50%);
  animation: pulse-ring 2s ease-out infinite;
}

@keyframes pulse-ring {
  0%   { transform: translate(-50%, -50%) scale(1); opacity: 1; }
  100% { transform: translate(-50%, -50%) scale(2); opacity: 0; }
}

/* Hover - activate all depth layers */
.depth-layers:hover::before,
.depth-layers:hover::after,
.depth-layers:hover .layer-bg,
.depth-layers:hover .layer-deep,
.depth-layers:hover .atmosphere,
.depth-layers:hover .parallax-indicator {
  opacity: 1;
}

.depth-layers:hover img {
  transform: translateZ(0) scale(1);
  filter: brightness(1.02) contrast(1.02);
}

/* Layer separation on hover */
.depth-layers:hover::before {
  transform: translateZ(100px) scale(0.9);
}

.depth-layers:hover::after {
  transform: translateZ(50px) scale(0.95);
}

.depth-layers:hover .layer-bg {
  transform: translateZ(-50px) scale(1.1);
}

.depth-layers:hover .layer-deep {
  transform: translateZ(-100px) scale(1.2);
}

/* JS cursor tracking (optional - adds .parallax-active class) */
.depth-layers.parallax-active {
  transition: transform 0.05s ease-out;
}

.depth-layers.parallax-active::before {
  transition: transform 0.05s ease-out;
}

.depth-layers.parallax-active::after {
  transition: transform 0.05s ease-out;
}

.depth-layers.parallax-active .layer-bg {
  transition: transform 0.05s ease-out;
}

.depth-layers.parallax-active .layer-deep {
  transition: transform 0.05s ease-out;
}

/* CSS-only fallback: automatic layer spread on hover */
.depth-layers:hover .parallax-indicator {
  opacity: 1;
}

/* Reduced motion */
@media (prefers-reduced-motion: reduce) {
  .depth-layers img,
  .depth-layers::before,
  .depth-layers::after,
  .depth-layers .layer-bg,
  .depth-layers .layer-deep,
  .depth-layers .atmosphere,
  .depth-layers .parallax-indicator {
    transition: opacity 0.3s ease;
  }
  .depth-layers:hover::before {
    transform: translateZ(80px) scale(0.92);
  }
  .depth-layers:hover::after {
    transform: translateZ(40px) scale(0.96);
  }
  .depth-layers:hover .layer-bg {
    transform: translateZ(-40px) scale(1.08);
  }
  .depth-layers:hover .layer-deep {
    transform: translateZ(-80px) scale(1.15);
  }
  .depth-layers .parallax-indicator::before {
    animation: none;
  }
}
