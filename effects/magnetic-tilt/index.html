CSS

/* Magnetic Tilt Effect - 3D tilt toward cursor with depth parallax layers */

.magnetic-tilt {
  position: relative;
  display: inline-block;
  overflow: hidden;
  border-radius: 12px;
  perspective: 1000px;
  transform-style: preserve-3d;
}

.magnetic-tilt img {
  display: block;
  width: 100%;
  height: auto;
  transform-style: preserve-3d;
  transition: transform 0.1s ease-out, filter 0.3s ease-out;
  will-change: transform;
}

/* Depth layers using pseudo-elements */
.magnetic-tilt::before,
.magnetic-tilt::after {
  content: '';
  position: absolute;
  inset: -15%;
  background: inherit;
  background-size: cover;
  background-position: center;
  transform-style: preserve-3d;
  pointer-events: none;
  border-radius: 12px;
  opacity: 0;
  transition: opacity 0.2s ease-out, transform 0.1s ease-out;
}

/* Layer 1: Background blur (furthest back) */
.magnetic-tilt::before {
  z-index: -2;
  filter: brightness(0.6) blur(12px);
  transform: translateZ(-60px) scale(1.2);
}

/* Layer 2: Mid-ground texture */
.magnetic-tilt::after {
  z-index: -1;
  filter: brightness(0.85) blur(4px);
  transform: translateZ(-30px) scale(1.1);
}

/* Glare/reflection overlay */
.magnetic-tilt .glare {
  position: absolute;
  inset: 0;
  background: linear-gradient(
    135deg,
    rgba(255,255,255,0) 0%,
    rgba(255,255,255,0.12) 45%,
    rgba(255,255,255,0.08) 50%,
    rgba(255,255,255,0.12) 55%,
    rgba(255,255,255,0) 100%
  );
  opacity: 0;
  pointer-events: none;
  transition: opacity 0.3s ease-out;
  mix-blend-mode: overlay;
  border-radius: 12px;
  z-index: 2;
}

/* Hover state */
.magnetic-tilt:hover::before,
.magnetic-tilt:hover::after,
.magnetic-tilt:hover .glare {
  opacity: 1;
}

.magnetic-tilt:hover img {
  filter: brightness(1.08) contrast(1.05) saturate(1.1);
}

/* Active tilt state (when JS adds .tilting class) */
.magnetic-tilt.tilting img,
.magnetic-tilt.tilting::before,
.magnetic-tilt.tilting::after {
  transition: transform 0.05s ease-out;
}

/* CSS-only fallback: subtle tilt on hover without cursor tracking */
.magnetic-tilt:hover img {
  transform: perspective(1000px) rotateX(2deg) rotateY(-2deg) scale(1.02);
}

.magnetic-tilt:hover::before {
  transform: translateZ(-60px) scale(1.2) perspective(1000px) rotateX(1.5deg) rotateY(-1.5deg);
}

.magnetic-tilt:hover::after {
  transform: translateZ(-30px) scale(1.1) perspective(1000px) rotateX(1deg) rotateY(-1deg);
}

/* Reduced motion */
@media (prefers-reduced-motion: reduce) {
  .magnetic-tilt img,
  .magnetic-tilt::before,
  .magnetic-tilt::after,
  .magnetic-tilt .glare {
    transition: none;
  }
  .magnetic-tilt:hover img {
    transform: scale(1.02);
  }
  .magnetic-tilt:hover::before {
    transform: translateZ(-60px) scale(1.2);
  }
  .magnetic-tilt:hover::after {
    transform: translateZ(-30px) scale(1.1);
  }
}
