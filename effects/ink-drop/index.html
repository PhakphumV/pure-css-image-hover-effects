/* Ink Drop Effect */

.ink-drop {
  position: relative;
  overflow: hidden;
}

.ink-drop::after {
  content: "";
  position: absolute;
  inset: 50%;
  width: 0;
  height: 0;
  border-radius: 50%;
  background: rgba(0, 0, 0, 0.1);
  transform: translate(-50%, -50%);
  transition: width 0.3s, height 0.3s;
}

.ink-drop:hover::after {
  width: 200%;
  height: 200%;
}
