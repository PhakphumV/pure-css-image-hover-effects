CSS

/* Glitch Slice Effect - Horizontal glitch slices with hue shifts */

.glitch-slice {
  position: relative;
  display: inline-block;
  overflow: hidden;
  border-radius: 8px;
}

.glitch-slice img {
  display: block;
  width: 100%;
  height: auto;
  transition: transform 0.3s ease-out, filter 0.3s ease-out;
}

/* Generate glitch slices using clip-path on pseudo-elements */
/* We'll use multiple layers with different clip-paths */
.glitch-slice::before,
.glitch-slice::after,
.glitch-slice .slice-1,
.glitch-slice .slice-2,
.glitch-slice .slice-3,
.glitch-slice .slice-4,
.glitch-slice .slice-5 {
  content: '';
  position: absolute;
  inset: 0;
  background: inherit;
  background-size: cover;
  background-position: center;
  opacity: 0;
  pointer-events: none;
  mix-blend-mode: difference;
  z-index: 1;
  transition: opacity 0.1s ease, transform 0.1s ease;
}

/* Slice positions - horizontal bands */
.glitch-slice::before   { clip-path: inset(0% 0 80% 0); }
.glitch-slice::after    { clip-path: inset(15% 0 65% 0); }
.glitch-slice .slice-1  { clip-path: inset(30% 0 50% 0); }
.glitch-slice .slice-2  { clip-path: inset(45% 0 35% 0); }
.glitch-slice .slice-3  { clip-path: inset(60% 0 20% 0); }
.glitch-slice .slice-4  { clip-path: inset(75% 0 5% 0); }
.glitch-slice .slice-5  { clip-path: inset(90% 0 0 0); }

/* Hue rotation for each slice */
.glitch-slice::before   { filter: hue-rotate(0deg) contrast(1.5) brightness(1.2); }
.glitch-slice::after    { filter: hue-rotate(60deg) contrast(1.5) brightness(1.2); }
.glitch-slice .slice-1  { filter: hue-rotate(120deg) contrast(1.5) brightness(1.2); }
.glitch-slice .slice-2  { filter: hue-rotate(180deg) contrast(1.5) brightness(1.2); }
.glitch-slice .slice-3  { filter: hue-rotate(240deg) contrast(1.5) brightness(1.2); }
.glitch-slice .slice-4  { filter: hue-rotate(300deg) contrast(1.5) brightness(1.2); }
.glitch-slice .slice-5  { filter: hue-rotate(360deg) contrast(1.5) brightness(1.2); }

/* Scanline overlay */
.glitch-slice .scanlines {
  position: absolute;
  inset: 0;
  background: repeating-linear-gradient(
    0deg,
    transparent,
    transparent 2px,
    rgba(0,0,0,0.1) 2px,
    rgba(0,0,0,0.1) 4px
  );
  opacity: 0;
  pointer-events: none;
  z-index: 2;
  animation: scanline-move 0.1s linear infinite;
  transition: opacity 0.2s ease;
}

@keyframes scanline-move {
  0% { background-position: 0 0; }
  100% { background-position: 0 4px; }
}

/* RGB shift lines */
.glitch-slice .rgb-lines {
  position: absolute;
  inset: 0;
  pointer-events: none;
  z-index: 3;
  opacity: 0;
  transition: opacity 0.2s ease;
}

.glitch-slice .rgb-lines::before,
.glitch-slice .rgb-lines::after {
  content: '';
  position: absolute;
  inset: 0;
  background: inherit;
  background-size: cover;
  background-position: center;
  mix-blend-mode: screen;
}

.glitch-slice .rgb-lines::before {
  filter: hue-rotate(-15deg);
  clip-path: inset(20% 0 70% 0);
  animation: rgb-shift 0.05s infinite alternate;
}

.glitch-slice .rgb-lines::after {
  filter: hue-rotate(15deg);
  clip-path: inset(60% 0 30% 0);
  animation: rgb-shift 0.05s infinite alternate-reverse;
}

@keyframes rgb-shift {
  0%   { transform: translateX(-2px); }
  100% { transform: translateX(2px); }
}

/* Hover - activate glitch */
.glitch-slice:hover::before,
.glitch-slice:hover::after,
.glitch-slice:hover .slice-1,
.glitch-slice:hover .slice-2,
.glitch-slice:hover .slice-3,
.glitch-slice:hover .slice-4,
.glitch-slice:hover .slice-5 {
  opacity: 1;
  animation: glitch-jitter 0.3s infinite;
}

.glitch-slice:hover .scanlines,
.glitch-slice:hover .rgb-lines {
  opacity: 1;
}

.glitch-slice:hover img {
  transform: scale(1.02);
  filter: brightness(1.1) contrast(1.05);
}

/* Jitter animation - each slice moves independently */
@keyframes glitch-jitter {
  0%   { transform: translateX(0); opacity: 1; }
  10%  { transform: translateX(-3px); opacity: 0.8; }
  20%  { transform: translateX(3px); opacity: 1; }
  30%  { transform: translateX(-2px); opacity: 0.9; }
  40%  { transform: translateX(2px); opacity: 1; }
  50%  { transform: translateX(0); opacity: 0.8; }
  60%  { transform: translateX(4px); opacity: 1; }
  70%  { transform: translateX(-4px); opacity: 0.9; }
  80%  { transform: translateX(1px); opacity: 1; }
  90%  { transform: translateX(-1px); opacity: 0.8; }
  100% { transform: translateX(0); opacity: 1; }
}

/* Staggered delay for each slice */
.glitch-slice:hover::before   { animation-delay: 0s; }
.glitch-slice:hover::after    { animation-delay: 0.02s; }
.glitch-slice:hover .slice-1  { animation-delay: 0.04s; }
.glitch-slice:hover .slice-2  { animation-delay: 0.06s; }
.glitch-slice:hover .slice-3  { animation-delay: 0.08s; }
.glitch-slice:hover .slice-4  { animation-delay: 0.1s; }
.glitch-slice:hover .slice-5  { animation-delay: 0.12s; }

/* Reduced motion */
@media (prefers-reduced-motion: reduce) {
  .glitch-slice:hover::before,
  .glitch-slice:hover::after,
  .glitch-slice:hover .slice-1,
  .glitch-slice:hover .slice-2,
  .glitch-slice:hover .slice-3,
  .glitch-slice:hover .slice-4,
  .glitch-slice:hover .slice-5 {
    animation: none;
    transform: translateX(0);
    opacity: 0.5;
  }
  .glitch-slice .scanlines,
  .glitch-slice .rgb-lines {
    animation: none;
  }
}
