.bounce-left {

  content: '';
  position: absolute;
  top: 0; left: 100%;
  width: 100%; height: 100%;

  transition: left 0.6s cubic-bezier(0.68, -0.55, 0.265, 1.55);
  z-index: 1;
}
.bounce-left:hover::before {
