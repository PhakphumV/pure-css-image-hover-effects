Liquid Morph

Organic blob shape morphs fluidly over image using animated border-radius on hover.

CSS

/* Liquid Morph Effect - Organic shape morphing with border-radius animation */

.liquid-morph {
  position: relative;
  display: inline-block;
  overflow: hidden;
  border-radius: 16px;
}

.liquid-morph img {
  display: block;
  width: 100%;
  height: auto;
  transition: transform 0.8s cubic-bezier(0.25, 0.46, 0.45, 0.94),
              filter 0.8s cubic-bezier(0.25, 0.46, 0.45, 0.94),
              border-radius 0.8s cubic-bezier(0.25, 0.46, 0.45, 0.94);
}

/* Liquid blob that morphs over the image */
.liquid-morph::before {
  content: '';
  position: absolute;
  inset: -30%;
  background: inherit;
  background-size: cover;
  background-position: center;
  opacity: 0;
  pointer-events: none;
  mix-blend-mode: overlay;
  transition: opacity 0.4s ease;
  z-index: 1;
  border-radius: 
    60% 40% 30% 70% / 60% 30% 70% 40%;
  animation: liquid-morph-1 8s ease-in-out infinite;
  filter: blur(4px) contrast(1.1) saturate(1.2);
}

/* Second liquid layer for depth */
.liquid-morph::after {
  content: '';
  position: absolute;
  inset: -25%;
  background: inherit;
  background-size: cover;
  background-position: center;
  opacity: 0;
  pointer-events: none;
  mix-blend-mode: soft-light;
  transition: opacity 0.4s ease;
  z-index: 2;
  border-radius: 
    40% 60% 70% 30% / 30% 70% 40% 60%;
  animation: liquid-morph-2 10s ease-in-out infinite reverse;
  filter: blur(2px) contrast(1.05);
}

/* Ripple overlay */
.liquid-morph .ripple {
  position: absolute;
  inset: 0;
  background: radial-gradient(
    ellipse at center,
    rgba(255,255,255,0.15) 0%,
    transparent 70%
  );
  opacity: 0;
  pointer-events: none;
  mix-blend-mode: screen;
  transition: opacity 0.4s ease;
  z-index: 3;
  border-radius: 16px;
}

/* Edge highlight */
.liquid-morph .edge {
  position: absolute;
  inset: 0;
  border: 2px solid transparent;
  border-radius: 16px;
  pointer-events: none;
  z-index: 4;
  opacity: 0;
  transition: opacity 0.4s ease, border-color 0.4s ease;
}

.liquid-morph .edge::before {
  content: '';
  position: absolute;
  inset: -2px;
  border-radius: 16px;
  background: linear-gradient(
    135deg,
    rgba(255,255,255,0.4),
    rgba(255,255,255,0.1),
    transparent,
    rgba(0,0,0,0.1)
  );
  -webkit-mask: 
    linear-gradient(#fff 0 0) content-box, 
    linear-gradient(#fff 0 0);
  -webkit-mask-composite: xor;
  mask-composite: exclude;
  opacity: 0;
  transition: opacity 0.4s ease;
}

/* Hover state */
.liquid-morph:hover::before,
.liquid-morph:hover::after {
  opacity: 0.6;
}

.liquid-morph:hover .ripple {
  opacity: 1;
  animation: ripple-pulse 2s ease-out infinite;
}

.liquid-morph:hover .edge {
  opacity: 1;
}

.liquid-morph:hover .edge::before {
  opacity: 1;
}

.liquid-morph:hover img {
  transform: scale(1.03);
  filter: brightness(1.05) contrast(1.03);
  border-radius: 
    55% 45% 35% 65% / 55% 35% 65% 45%;
}

/* Morphing animations */
@keyframes liquid-morph-1 {
  0%, 100% {
    border-radius: 60% 40% 30% 70% / 60% 30% 70% 40%;
    transform: rotate(0deg) scale(1);
  }
  25% {
    border-radius: 50% 50% 50% 50% / 70% 30% 60% 40%;
    transform: rotate(2deg) scale(1.02);
  }
  50% {
    border-radius: 40% 60% 70% 30% / 50% 50% 50% 50%;
    transform: rotate(-1deg) scale(1.01);
  }
  75% {
    border-radius: 70% 30% 40% 60% / 30% 70% 40% 60%;
    transform: rotate(1deg) scale(1.02);
  }
}

@keyframes liquid-morph-2 {
  0%, 100% {
    border-radius: 40% 60% 70% 30% / 30% 70% 40% 60%;
    transform: rotate(0deg) scale(1);
  }
  25% {
    border-radius: 30% 70% 60% 40% / 60% 40% 70% 30%;
    transform: rotate(-2deg) scale(1.015);
  }
  50% {
    border-radius: 70% 30% 40% 60% / 40% 60% 30% 70%;
    transform: rotate(1deg) scale(1.01);
  }
  75% {
    border-radius: 60% 40% 30% 70% / 70% 30% 60% 40%;
    transform: rotate(-1deg) scale(1.015);
  }
}

@keyframes ripple-pulse {
  0%   { transform: scale(1); opacity: 0.3; }
  50%  { transform: scale(1.1); opacity: 0.1; }
  100% { transform: scale(1); opacity: 0.3; }
}

/* Interactive hover morph - different path */
.liquid-morph:hover::before {
  animation: liquid-morph-hover-1 1.2s cubic-bezier(0.25, 0.46, 0.45, 0.94) forwards;
}

.liquid-morph:hover::after {
  animation: liquid-morph-hover-2 1.2s cubic-bezier(0.25, 0.46, 0.45, 0.94) forwards;
}

  }
  100% {
    border-radius: 55% 45% 35% 65% / 55% 35% 65% 45%;
    transform: rotate(0deg) scale(1.05);
  }

@keyframes liquid-morph-hover-2 {

    border-radius: 40% 60% 70% 30% / 30% 70% 40% 60%;
