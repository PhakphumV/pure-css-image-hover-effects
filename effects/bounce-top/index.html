CSS

.bounce-top {
  overflow: hidden;
  display: inline-block;
  border-radius: 8px;
  position: relative;
}
.bounce-top img {
  display: block;
  width: 100%;
  height: auto;
}
.bounce-top::before {
  content: '';
  position: absolute;
  top: -100%; left: 0;
  width: 100%; height: 100%;
  background: white;
  transition: top 0.6s cubic-bezier(0.68, -0.55, 0.265, 1.55);
  z-index: 1;
}
.bounce-top:hover::before {
  top: 100%;
}
